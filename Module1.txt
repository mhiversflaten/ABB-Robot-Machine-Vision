        ! Flytt over puck og sett ready_flag:=1; Ta bilde og fortsett:
        ! movePuckToMiddle puck_position;
    ENDPROC

    PROC closeGripper(bool state)
        WaitTime 0.1;
        IF state=TRUE THEN
          setDO PneumaticGripper, 1;
        ELSEIF state=FALSE THEN
          setDO PneumaticGripper, 0;
        ENDIF
        WaitTime 0.2;
    ENDPROC
